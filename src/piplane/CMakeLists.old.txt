cmake_minimum_required(VERSION 3.13)

# initialize the SDK based on PICO_SDK_PATH
# note: this must happen before project()
include(pico_sdk_import.cmake)

project(dshot C ASM)

# initialize the Raspberry Pi Pico SDK
pico_sdk_init()

add_library(dshot INTERFACE)
pico_generate_pio_header(dshot ${CMAKE_CURRENT_LIST_DIR}/dshot.pio)

target_sources(dshot INTERFACE ${CMAKE_CURRENT_LIST_DIR}/dshot.c)
target_include_directories(dshot INTERFACE ${CMAKE_CURRENT_LIST_DIR}/include)
target_link_libraries(dshot INTERFACE
    pico_stdlib
    hardware_pio
    microros
)

include_directories(${CMAKE_CURRENT_LIST_DIR}/include)

# Add the piplane executable
add_executable(piplane ${CMAKE_CURRENT_LIST_DIR}/piplane.c)

# Link the necessary libraries to the piplane executable
target_link_libraries(piplane dshot pico_stdlib hardware_pio)

add_subdirectory(examples EXCLUDE_FROM_ALL)
